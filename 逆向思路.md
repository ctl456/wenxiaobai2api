```json
//signature

_createHmacHelper: function (e) {

        return new m.HMAC.init(e, r).finalize(t)
```

```python
import hmac
import hashlib
import base64

key = "TkoWuEN8cpDJubb7Zfwxln16NQDZIc8z" #固定值
message = "x-date: Sun, 02 Mar 2025 08:44:41 GMT\ndigest: SHA-256=47DEQpj8HBSa+/TImW+5JCeuQeRkm5NMpJWZG3hSuFU="

hmac_result = hmac.new(
    key.encode('utf-8'),
    message.encode('utf-8'),
    hashlib.sha1
).hexdigest()
```
